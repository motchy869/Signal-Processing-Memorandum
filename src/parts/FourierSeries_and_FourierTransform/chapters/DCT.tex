\chapter{離散Cosine変換(DCT)}
    \section{DCT基底の正規直交性を導出するための補題}
        \begin{shadebox}
            $M\in\integers,\;N\in\naturalNumbers$とする。
            さらに$M$は$2N$の倍数ではないとする。
            このとき次式が成り立つ。
            \[ \sum_{n=0}^{N-1} \cos\frac{M(n+1/2)}{N}\pi = 0\]
        \end{shadebox}
        \begin{proof}
            (証明1 複素指数関数を用いる証明)
            \par\noindent
            (証明2 複素数を用いない証明(所謂「望遠鏡和」を使う))
        \end{proof}
